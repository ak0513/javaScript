테이블 생성

객체나 배열의 데이터를 table을 파싱 할 수 있다.

배열로 생성

인덱스 프로퍼티1 프로퍼티2 프로퍼티3
인덱스 프로퍼티1 프로퍼티2 프로퍼티3
// 배열 생성 var table = [ { property1: "값1-1", property2: "값1-2", property3: "값1-3" }, { property1: "값2-1", property2: "값2-2", property3: "값2-3" }, { property1: "값3-1", property2: "값3-2", property3: "값3-3" } ]; // 테이블에 배열 정보 추가 var tableBody = document.querySelector('#objectArrayTable tbody'); table.forEach(function(obj, index) { var row = tableBody.insertRow(); var indexCell = row.insertCell(0); var property1Cell = row.insertCell(1); var property2Cell = row.insertCell(2); var property3Cell = row.insertCell(3); indexCell.innerHTML = index; property1Cell.innerHTML = obj.property1; property2Cell.innerHTML = obj.property2; property3Cell.innerHTML = obj.property3; });

// 객체 생성 var tableArr = { property1: "값1", property2: "값2", property3: "값3" }; // 테이블에 객체 정보 추가 var arrTableBody = document.querySelector('#objectTable tbody'); for (var key in tableArr) { if (tableArr.hasOwnProperty(key)) { var row = arrTableBody.insertRow(); var propertyNameCell = row.insertCell(0); var propertyValueCell = row.insertCell(1); propertyNameCell.innerHTML = key; propertyValueCell.innerHTML = tableArr[key]; } }
