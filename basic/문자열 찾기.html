문자열 찾기

indexOf

indexOf 문자열 또는 배열에서 특정 요소나 문자열이 처음으로 나타나는 위치를 찾는 함수입니다.
주어진 요소나 문자열이 존재하지 않는 경우에는 보통 -1을 반환합니다.

문자열에서 인덱스 찾기

let text = "Hello"; let numbers = [10, 20, 30, 40, 50]; // 문자열에서 인덱스 찾기 text.indexOf("e"); // 출력: 1 // 존재하지 않는 문자의 경우 -1 반환 text.indexOf("x"); // 출력: -1 // 시작 인덱스 지정하여 검색하기(3번 인덱스부터 검색) text.indexOf("l", 3); // 출력: 3(2번 인덱스의 l은 제외) // 문자열 내에 동일 문자가 여러 번 나오는 경우 text.indexOf("l"); // 출력: 2 (첫 번째 'l'의 인덱스) // 대소문자를 구분 text.indexOf("h"); // 출력: -1 text.indexOf("H"); // 출력: 0 // 배열에서 요소의 인덱스 찾기 numbers.indexOf(30); // 출력: 2

문자열 내의 특정 문자가 있는지 확인하는 방법

let text = 'Hello'; if(text.indexOf('H') !== -1) { console.log('true'); } else { console.log('false'); } // true

문자열 내의 특정 문자 개수 구하기

let text = 'hello world'; function getIndexOfNum(text, searchValue) { let count = 0; let pos = text.indexOf(searchValue); while(pos !== -1) { pos = text.indexOf(searchValue, pos + 1); count++; } return count; } getIndexOfNum(text, o); // 2 getIndexOfNum(text, l); // 3

문자열의 n번째 위치 index값 가져오기

let text = 'hello wolrd hello wolrd'; function getIndexOfPos(text, searchValue, nth) { let count = 0; let pos = 0; while(count < nth && pos !== 1) { if(text.indexOf(searchValue) === 0) { if(nth === 1) { pos = 1; break; } count = 1; } pos = text.indexOf(searchValue, pos + 1); count++; } return pos; } getIndexOfPos(text, 'h' 2); // 12 getIndexOfPos(text, 'h' 1); // 1
