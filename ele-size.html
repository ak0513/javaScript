element.offsetWidth/offsetHeight

  • padding, border 를 포함한 콘텐츠의 사이즈
  • margin 제외
margin - - - -
border 3 3 3 3
padding 10 10 10 10
200x200
var box = document.querySelector('.box'); box.offsetWidth // 226 = width(200) + padding(10+10) + border(3+3) // padding, border 를 포함한 컨텐츠의 너비 / margin 제외 box.offsetHeight // 226 = height(200) + padding(10+10) + border(3+3) // padding, border 를 포함한 컨텐츠의 높이 / margin 제외

element.clientWidth/clientHeight

  • padding 을 포함한 콘텐츠의 사이즈
  • margin, border 제외
margin - - - -
border 3 3 3 3

200x200
box.clientWidth // 220 = height(200) + padding(10+10) // padding 을 포함한 컨텐츠의 너비 / margin, border 제외 box.clientHeight // 220 = height(200) + padding(10+10) // padding 을 포함한 컨텐츠의 높이 / margin, border 제외

element.scrollWidth/scrollHeight

  • padding 을 포함한 콘텐츠의 사이즈(자식요소 포함)
  • margin, border 제외
box.scrollWidth // 320 = padding(10+10) + box-scroll-x(300) // padding 을 포함한 컨텐츠의 너비(자식요소 포함) box.scrollHeight // 385 = padding(10+10) + box-scroll-x(55) + margin(10) + box-scroll-y(300) // padding 을 포함한 컨텐츠의 높이(자식요소 포함)

element.getBoundingClientRect().width/height

  • padding, border 를 포함한 콘텐츠의 렌더링 된 사이즈
  • margin 제외
// offsetWidth, offsetHeight와 동일 box.getBoundingClientRect().width // 226 = width(200) + padding(10+10) + border(3+3) box.getBoundingClientRect().height // 226 = height(200) + padding(10+10) + border(3+3) // box에 transform:scale(50%); 적용 시 box.getBoundingClientRect().width // 113(transform:scale(50%)) box.getBoundingClientRect().height // 113(transform:scale(50%))
