node(노드) 탐색

NodeList로 반환

var self = document.querySelector('#self'); // <div id="self">...</div> var parentNode = self.parentNode; // <div id="parent">...</div> var firstChild = self.firstChild; // self var lastChild = self.lastChild; // #text var childNodes = self.childNodes; // NodeList(5) var previousSibling = self.previousSibling; // #text var nextSibling = self.nextSibling; // #text

element(요소) 탐색

HTMLCollection으로 반환

var self = document.querySelector('#self'); // <div id="self">...</div> var parentElement = self.parentElement; // <div id="parent">...</div> var firstElementChild = self.firstElementChild; // <div id="child1">...</div> var lastElementChild = self.lastElementChild; // <div id="child2">...</div> var children = self.children; // HTMLCollection(2) var previousElementSibling = self.previousElementSibling; // <div id="firstChild">...</div> var nextElementSibling = self.nextElementSibling; // <div id="lastChild">...</div>

.box {width:200px;height:200px;padding:10px;border:3px solid #000;box-sizing: content-box;} .box-inner {margin:10px;padding:10px;border:3px solid #000;} .box-inner.box-scroll-x {width: 300px;} .box-inner.box-scroll-y {height: 300px;margin-top: 10px;}

element.offsetWidth/offsetHeight

var box = document.querySelector('.box'); box.offsetWidth // 226 = width(200) + padding(10+10) + border(3+3) // padding, border 를 포함한 컨텐츠의 너비 / margin 제외 box.offsetHeight // 226 = height(200) + padding(10+10) + border(3+3) // padding, border 를 포함한 컨텐츠의 높이 / margin 제외

element.clientWidth/clientHeight

box.clientWidth // 220 = height(200) + padding(10+10) // padding 을 포함한 컨텐츠의 너비 / margin, border 제외 box.clientHeight // 220 = height(200) + padding(10+10) // padding 을 포함한 컨텐츠의 높이 / margin, border 제외

element.scrollWidth/scrollHeight

box.scrollWidth // 320 = padding(10+10) + box-scroll-x(300) // padding 을 포함한 컨텐츠의 너비(자식요소 포함) box.scrollHeight // 385 = padding(10+10) + box-scroll-x(55) + margin(10) + box-scroll-y(300) // padding 을 포함한 컨텐츠의 높이(자식요소 포함)

element.getBoundingClientRect().width/height

// offsetWidth, offsetHeight와 동일 box.getBoundingClientRect().width // 226 = width(200) + padding(10+10) + border(3+3) box.getBoundingClientRect().height // 226 = height(200) + padding(10+10) + border(3+3) // box에 transform:scale(90%); 적용 시 box.getBoundingClientRect().width // 203.39999389648438(transform:scale(90%)) box.getBoundingClientRect().height // 203.39990234375(transform:scale(90%))
